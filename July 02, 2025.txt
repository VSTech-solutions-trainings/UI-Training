		var firstName = "MAxM" // maxm // MAxm // MAXm

		//built in methods of String
		var res = firstName.length // This will log the length of the string
		var res = firstName.charAt(1) // This will log 'a', it will take only the positive values
		var res = firstName.at(-3) // same as charAt but this will accept negative value or index
		var res = firstName.indexOf('p') // if it didn't find the chat it will return -1
		var res = firstName.includes('o') // internally uses indexOf - -1 return false, else return true
		var res = firstName.toUpperCase() // converts into uppercase - MAX
		var res = firstName.toLowerCase() // converts into uppercase - max
		var res = firstName.replace('M', 'T') // This will replace the first occurence of the char
		var res = firstName.replaceAll('M', 'T') // This will replace all the occureneces of the char
		var res = firstName.lastIndexOf('A') // This will find out the last index of the char
